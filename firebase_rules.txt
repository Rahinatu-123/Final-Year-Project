rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow anyone to read users
    match /users/{userId} {
      allow read: if true;
      allow create: if request.auth != null && request.auth.uid == userId;
      // Allow users to update their own profile including business fields
      allow update: if request.auth != null && request.auth.uid == userId;

      // Connections subcollection: user can create/delete their own connections
      match /connections/{targetUid} {
        allow create: if request.auth != null 
                      && request.auth.uid == userId;
        allow delete: if request.auth != null && request.auth.uid == userId;
        allow read: if request.auth != null;
      }

      // Followers subcollection: profile owner can read, follower can create/delete
      match /followers/{followerId} {
        allow create: if request.auth != null 
                      && request.auth.uid == followerId;
        allow delete: if request.auth != null && request.auth.uid == followerId;
        allow read: if request.auth != null;
      }
    }
    }

    // Posts collection: anyone can read, authenticated users can create/edit their own
    match /posts/{postId} {
      allow read: if true;
      allow create: if request.auth != null && request.auth.uid == request.resource.data.userId;
      allow update, delete: if request.auth != null && request.auth.uid == request.resource.data.userId;
    }

    // Orders collection: authenticated users can read/write their own orders
    match /orders/{orderId} {
      allow read: if request.auth != null && 
                  (request.auth.uid == resource.data.userId || 
                   request.auth.uid == resource.data.tailorId);
      allow create: if request.auth != null;
      allow update: if request.auth != null && 
                  (request.auth.uid == resource.data.userId || 
                   request.auth.uid == resource.data.tailorId);
      allow delete: if request.auth != null && 
                  (request.auth.uid == resource.data.userId || 
                   request.auth.uid == resource.data.tailorId);
    }

    // Default: deny all other access
    match /{document=**} {
      allow read, write: if false;
    }
  }
